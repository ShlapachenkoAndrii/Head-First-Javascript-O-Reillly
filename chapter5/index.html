            function alertSeatsAvailability() {
                //true is for available seats
                var seats = [true, false, true, false, false, true, false];
                for (var i = 0; i < seats.length; i++){
                    if(seats[i])
                        alert('Seat ' + (i+1) + ' is available');
                    else
                        alert('Seat ' + (i+1) + ' is unavailable');
                }
            };
